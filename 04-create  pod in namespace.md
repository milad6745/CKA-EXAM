# create  pod in namespace


```
└─# kubectl create namespace testns
namespace/testns created


└─# kubectl run test03 --image=nginx -n testns
pod/test03 created


└─# kubectl get ns
NAME                 STATUS   AGE
testns               Active   28s


└─# kubectl get pod -n testns
NAME     READY   STATUS    RESTARTS   AGE
test03   1/1     Running   0          39s
```
